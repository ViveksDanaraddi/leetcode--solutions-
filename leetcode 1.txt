class Solution {
    public boolean isPalindrome(int x) {
        if(x<0)
        {
            return false;
        }
        int n = x;
        int Reveres_num = 0;
         while(n>0)
         {
            int d = n%10;
            Reveres_num = Reveres_num*10 + d;
            n = n/10;
         }
         if(Reveres_num == x)
         {
            return true;
         }
        else
        {
            return false;
        }
        
        
    }
}
